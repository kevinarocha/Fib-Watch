                        // generates the fibonacci sequence in order when called upon
                        var fibGenerator = function*() {
                            let n1=0,n2=1,i=0;
                            while(true){
                                if(i==0){
                                    i++;
                                    yield 0;
                                }
                                if(i==1){
                                    i++;
                                    yield 1;
                                }
                                let ans = n1 + n2;
                                n1 = n2;
                                n2 = ans;
                                yield ans;
                            }
                         };
                        
                        
                        
                        // generate numbers from startvalue to the endValue
                        //logic functions(s)
                        function generateNumbers(eValue){
                            let numbers = [];
                            const gen = fibGenerator();
                            // we want to get all numbers from start to end
                            
                            for(let index = 0; index <= eValue; index++){
                                // this will execute in a loop until index = eValue
                                
                                //numbers.push(index);
                                numbers.push(gen.next().value);
                            }
                            
                            return numbers;
                        }
                        
                        
                        
                        //display or view functions

                            document.getElementById("results").innerHTML = templateRows;
                        
                        }

The Code is structured in four functions.

getValues

getValues is a function that gets the users value to stop displaying the fibbonacci sequence. It disallows any letters in the input. It generates the numbers based on the amount of numbers inputted by the user and calls the displayNumbers method after calling the generateNumbers method.

fibGenerator

fibGenerator is a function that mathematically calculates the fibbonacci sequence in order and returns the numbers individually everytime it is called upon.

generateNumbers

generateNumbers is a function that sets an array and generates the fibbonacci sequence by calling the fibGenerator method and placing it into the array and loops until the loop reaches the same number as the user inputted.

displayNumbers

displayNumbers is a function that loops by the size of numbers requested by the user and displays each number inside the array in string format.
